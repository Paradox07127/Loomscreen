```swift
private var hasFinishedFirstNavigation = false
private var firstNavigationContinuations: [CheckedContinuation<Void, Never>] = []

func awaitFirstNavigation(timeout: Duration) async -> Bool {
    if hasFinishedFirstNavigation { return true }
    return await withTaskGroup(of: Bool.self) { group in
        group.addTask { [weak self] in
            await withCheckedContinuation { cont in
                self?.firstNavigationContinuations.append(cont)
            }
            return true
        }
        group.addTask {
            try? await Task.sleep(for: timeout)
            return false
        }
        let result = await group.next() ?? false
        group.cancelAll()
        return result
    }
}
```

### Step 8: 新增 `WallpaperTransitionKind` 枚举

```swift
import Foundation

/// Visual transition profile chosen by the caller of
/// `ScreenManager.transitionSession(...)`. Maps to a duration
/// (per Gemini UX matrix in plan synthesis).
enum WallpaperTransitionKind {
    case userInitiated   // 0.30s — toolbar tab switch, manual select
    case automatic       // 0.80s — playlist auto-rotation
    case schedule        // 0.50s — schedule slot enter/exit
    case reload          // 0.10s — reload button (same wallpaper refresh)

    var duration: TimeInterval {
        switch self {
        case .userInitiated: return 0.30
        case .automatic:     return 0.80
        case .schedule:      return 0.50
        case .reload:        return 0.10
        }
    }
}
```

### Step 9: 新增 `ScreenManager.transitionSession` 方法
**文件**：`LiveWallpaper/ScreenManager.swift`

```swift
@MainActor
private func transitionSession(
    for screen: Screen,
    kind: WallpaperTransitionKind = .userInitiated,
    readinessTimeout: Duration = .seconds(5),
    makeNewSession: @MainActor () async throws -> (any WallpaperRuntimeSession)?
) async {
    let generation = bumpTransition(for: screen.id)

    // 1. Build new session (window orderBack with alphaValue = 0)
    let newSession: any WallpaperRuntimeSession
    do {
        guard let s = try await makeNewSession() else { return }
        newSession = s
    } catch {
        Logger.error("transitionSession: makeNewSession failed: \(error.localizedDescription)", category: .screenManager)
        return
    }

    // 2. Verify still current after async work
    guard isCurrentTransition(generation, for: screen.id) else {
        newSession.cleanup()
        return
    }

    // 3. Hide new window before fade-in
    newSession.wallpaperWindow?.alphaValue = 0
    newSession.wallpaperWindow?.orderBack(nil)

    // 4. Wait for readiness
    _ = await newSession.prepareForDisplay(timeout: readinessTimeout)
    guard isCurrentTransition(generation, for: screen.id) else {
        newSession.cleanup()
        return
    }

    // 5. Stage swap (returns old session — we'll cleanup after fade)
    let oldSession = screen.stageRuntimeSessionForTransition(newSession)

    // 6. Determine fade duration (Reduce Motion = 0)
    let reduceMotion = NSWorkspace.shared.accessibilityDisplayShouldReduceMotion
    let duration = reduceMotion ? 0 : kind.duration

    // 7. Crossfade
    if duration > 0 {
        await withCheckedContinuation { (continuation: CheckedContinuation<Void, Never>) in
            NSAnimationContext.runAnimationGroup { context in
                context.duration = duration
                context.timingFunction = CAMediaTimingFunction(name: .easeInEaseOut)
                newSession.wallpaperWindow?.animator().alphaValue = 1
                oldSession?.wallpaperWindow?.animator().alphaValue = 0
            } completionHandler: {
                continuation.resume()
            }
        }
    } else {
        newSession.wallpaperWindow?.alphaValue = 1
        oldSession?.wallpaperWindow?.alphaValue = 0
    }

    // 8. Cleanup old session AFTER visual handoff
    oldSession?.cleanup()
    notifyWallpaperSessionChanged()
}
```
